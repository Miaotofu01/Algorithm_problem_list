# B 双碳战略

```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
const int mod = 998244353;
LL qpow(LL a, int b)
{
    LL res = 1 % mod;
    for (; b; b >>= 1, a = a * a % mod)
        if (b & 1)
            res = res * a % mod;
    return res;
}
inline void solve()
{
    int n = 2026;
    vector<int> dp(n + 1);
    for (int i = 1; i < dp.size(); i++)
    {
        dp[i] = (2 * dp[i - 1] + qpow(2, i - 1)) % mod;
    }
    cout << dp[n] << endl;
}
signed main(){
    int T = 1;
    // cin >> T;
    while(T--){
        solve();
    }
}
```

# D 蓝桥竞技

观察题目易得
- 总人数一定是5的倍数
- 每个种类的最大人数一定是 总人数/5

只要满足上述条件，一定可以通过以下方式分配完成：
- 每次仅选取当前剩余人数最多的五个种类成团

故仅需检查以上两个条件即可

```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
void solve(){
    int n;
    cin >> n;
    vector<int> a(n);
    for (int &e : a)
        cin >> e;
    int sum = accumulate(a.begin(),a.end(), 0LL);
    if (sum % 5 != 0)
    {
        cout << "F" << endl;
        return;
    }
    int maxcnt = sum / 5;
    for (int e : a)
    {
        if (e > maxcnt)
        {
            cout << "F" << endl;
            return;
        }
    }
    cout << "T" << endl;
}
signed main(){
    int T = 1;
    cin >> T;
    while(T--){
        solve();
    }
}
```
# E LQ聚合

最优策略必定是前面放一部分L，后面放一部分R，不可能是LR相间地放
故采用枚举策略
- 先假设全设置为R，计算答案
- 依次枚举"当前位置及以前设置为L，以后设置为R"的答案，对于每次枚举：
	- 减去当前位置作为R的贡献
	- 加上当前位置作为L的贡献
	- 即可完成一次更新
- 记录枚举出的最优解即可
```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
void solve(){
    /*
    一定是前面放L,后面放R，从左到右枚举（当前位置以前放L,以后放R）的答案即可
    */
    int n;
    cin >> n;
    string s;
    cin >> s;
    int ans = 0; 
    int cntR = 0;
    for (int i = s.size() - 1; i >= 0; i--) // 假设全放R的答案
    {
        if (s[i] == 'L')
            ans += cntR;
        else
            cntR++;
    }
    int res = ans;
    for (int i = 0, cntL = 0; i < s.size(); i++)
    {
        if (s[i] == '?')
        {
            ans -= cntL; // 减去这个位置原先作为R的贡献
            ans += --cntR; // 加上这个位置作为L的贡献
            res=max(res,ans);
        }
        cntL += s[i] != 'R';
    }
    cout << res << endl;
}
signed main(){
    int T = 1;
    // cin >> T;
    while(T--){
        solve();
    }
}
```

# F 应急布线

可知树是边最少的连通图，所以考虑到树的性质，将每个连通块看作树的一个节点，构造生成树

树的边即为跳线数，故 **最少需要添加的应急跳线数量 = 树节点树-1 = 连通块个数 - 1**

接下来求接入跳线数量最大值的最小可能值。
- 先考虑特殊情况，已经全连通的情况下（连通块个数=1），显然不需要添加额外边 ，ans=0
- 接着，观察到，要使得ans=1
	- 连通块内每个点只能连接一条额外边，故一个连通块的最大度数为连通块内点的大小。宏观来看，整个图的最大度数为`n`。
	- 连成树所需的最小度数为`树边数*2 = (节点数-1)/2 = (连通块个数-1)/2`
	- 所以，只要整个图的最大度数>=连成树所需的度数 (`n>=(连通块个数-1)/2`)，即可使得`ans1=1`
- 最后，若不满足上述条件，必然会有某个点连接了两条额外边，即**ans=2**，可以证明，不会有更差的情况（因为最劣可以将连通块连成一条链来构造树）。

```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
struct DSU // 并查集范围[1,n]
{
    vector<int> f, sz;
    int bg = 1;
    DSU(int n)
    {
        f.resize(n + 1);
        iota(f.begin(), f.end(), 0);
        sz.assign(n + 1, 0);
    }
    int find(int x)
    {
        return f[x] == x ? x : (f[x] = find(f[x]));
    }
    bool same(int x, int y)
    {
        x += 1 - bg;
        y += 1 - bg;
        return find(x) == find(y);
    }
    bool merge(int x, int y)
    { // 把x合并到y的集合上
        x += 1 - bg;
        y += 1 - bg;
        x = find(x), y = find(y);
        if (x == y)
            return false;
        sz[y] += sz[x];
        f[x] = y;
        return true;
    }
    int cnt() // 计算连通块的数量O(n)
    {
        int res = 0;
        for (int i = 1; i < f.size(); i++)
            res += (f[i] == i);
        return res;
    }
    int size(int x)
    {
        x += 1 - bg;
        return sz[find(x)];
    }
};
void solve()
{
    int n, m;
    cin >> n >> m;

    DSU dsu(n);
    for (int i = 0; i < m; i++)
    {
        int u, v;
        cin >> u >> v;
        dsu.merge(u, v);
    }
    
    cout << (dsu.cnt() - 1) << " ";
    if (dsu.cnt() == 1)
        cout << 0 << endl;
    else if ((dsu.cnt() - 1) * 2 <= n)
        cout << 1 << endl;
    else
        cout << 2 << endl;
}
signed main(){
    int T = 1;
    // cin >> T;
    while(T--){
        solve();
    }
}
```

# G 理想温度

将数组`a`与数组`b`做差，得到数组`d`，不做任何处理时，答案`ans = d中0的数量`

**我们只关注数组d**，假设选定一个区间`[l,r]`，区间内频率最高的数字的出现频率为`f(l,r)` ，区间内0的个数为`cnt0(l,r)`
那么这段区间对答案的影响为 `q(l,r) = f(l,r) - cnt0(l,r)` 即答案为 `ans+q(l,r)`

那么题目就转化为求出最大的 `q(l,r) = f(l,r) - cnt0(l,r)`
- `cnt0`可以通过前缀和维护
- 对于`f`，可以发现，一个数作为出现频率的最大的数的时机，是其作为右端点的时候，所以，**我们只关心右端点的数字**

- 所以可以定义`dp[i]`为：最大的`q(j,i) (j<=i)` , 所以答案为  $ans+dp_{max}$ 我们的目的是找到每一个`i`对应的`j`
- 从左到右遍历，维护 `pre[x]`: x上一次出现的位置
- 我们考虑一个右端点`d[i]`,这个端点有两种情况
	1. 接在前一个`d[i]`作为右端点的区间后面 `dp[i] = dp[pre[d[i]]-cnt0(pre[d[i]],i) + 1`
	2. `d[i]`作为一个独立区间 `dp[i]=1`
所以，状态转移方程：`dp[i] = max(1 ,dp[pre[d[i]]-cnt0(pre[d[i]],i) + 1)`

```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
inline void solve()
{
    /*
    对于每一个d，要么连接上一个d，要么自己作区间
    */
    int n;
    cin >> n;
    vector<int> a(n + 1), b(n + 1), d(n + 1), cnt0(n + 1);
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    for (int i = 1; i <= n; i++)
    {
        cin >> b[i];
        d[i] = b[i] - a[i];
        cnt0[i] = (d[i] == 0) + cnt0[i - 1];
    }
    int ans = 0;
    map<int, pair<int,int>> pre;//first维护贡献，second维护位置
    for (int i = 1; i <= n; i++)
    {
        if (!d[i])
            continue;
        int res = max(1LL, 1 + pre[d[i]].first - (cnt0[i] - cnt0[pre[d[i]].second]));
        pre[d[i]] = {res, i};
        ans=max(ans, res);
    }
    cout << ans + cnt0[n] << endl;
}
signed main(){
    int T = 1;
    // cin >> T;
    while(T--){
        solve();
    }
}
```

# H 足球训练

考虑所有b均等于1的情况，要使得一次操作令$a_1*a_2*...*a_n$的增量最大化，如何选取？

应该选取最小的$a_i$，为什么？

假设选取a_i，增量为$d=\frac{\prod_{j=1}^n a_j}{a_i}$ 要使得d最大化，只需选取最小的a_i

那么把数组b考虑进来，又会发生什么？

增量应修改为$d=\frac{\prod_{j=1}^n a_j}{a_i}*b_i$ ，那么每次应选取最大的增量，即选取最大的$\frac{b_i}{a_i}$

那么经历完m次修改后，我们应最大化最小的增量，采用浮点数二分的方法来确定这个值

确定了最大的最小增量d后，即可确定每个元素应该增加的次数$k_i$，$k_i$的计算如下

$$
d = \frac {a_i + b_i * k_i}{b_i} = \frac{a_i}{b_i} + k_i \
$$
可以得到
$$
k_i = d - \frac {a_i} {b_i} (k_i>=0) 
$$
```cpp
#include<bits/stdc++.h>
using namespace std;
using LL = long long;
#define int LL
const int MOD = 998244353;
inline void solve()
{
    int n, m;
    cin >> n >> m;
    vector<int> a(n), b(n);
    for (int i = 0; i < n; i++)
    {
        cin >> a[i] >> b[i];
    }
    // 浮点数二分
    double l = 0, r = 1e14;
    const double eps = 1e-8; // eps 表示精度，取决于题目对精度的要求
    while (r - l > eps)
    {
        double mid = (l + r) / 2;
        int k = 0;
        for (int i = 0; i < n; i++)
            k += ceil(max(0.0, mid - 1.0 * a[i] / b[i]));
        if (k <= m)
            l = mid;
        else
            r = mid;
    }
    int ans = 1;
    vector<int> c(n);
    for (int i = 0; i < n; i++)
    {
        int k = ceil(max(0.0, l - 1.0 * a[i] / b[i]));
        m -= k;
        c[i] = a[i] + k * b[i];
    }
    for (int i = 0; i < n; i++)//分配剩余的操作次数
    {
        if (m && abs(1.0 * c[i] / b[i] - l) <= eps)
        {
            c[i] += b[i];
            m--;
        }
        (ans *= c[i]) %= MOD;
    }
    cout << ans << endl;
}
signed main(){
    int T = 1;
    // cin >> T;
    while(T--){
        solve();
    }
}
```
